# classify "point" cities
us_cities <- us_cities |>
  mutate(on_point = str_detect(NAME, "Point") |
           str_detect(NAME, "point")) |>
  mutate(on_point_class = ifelse(on_point,
                                 "contains 'point'",
                                 "a different city")) |>
  filter(PopPlLat > 20) |>
  filter(PopPlLong > -130) |>
  filter(PopPlLong < -60)

# subset (useful for labeling later)
point_cities <- us_cities |>
  filter(on_point) |>
  filter(POP_2010 > 1e4)
# make rough map
us_cities |>
  ggplot() +
  geom_sf()

# make map!
